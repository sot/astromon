[docs]def binned_median(years, vals, bins_per_year=2):
    """Compute the median value in each bin of the input data.

    TODO: return bin edges and centers.
    """
    years_bin = np.trunc(years * bins_per_year) / bins_per_year
    t = Table([years_bin, years, vals], names=['years_bin', 'years', 'vals'])
    tg = t.group_by('years_bin')
    tga = tg.groups.aggregate(np.median)
    return tga['years'], tga['vals']




[docs]def get(obsids, start, stop, msids=PERISCOPE_MSIDS):
    """
    Get a collection of MSIDs for each OBSID.
    """
    dat = fetch.MSIDset(msids, start, stop, stat='5min')

    normal_suns.interval_pad = (200, 100000)
    safe_suns.interval_pad = (200, 200000)
    for msid in dat:
        dat[msid].remove_intervals(normal_suns | safe_suns)
        # what are these???
        dat[msid].remove_intervals([('2016:064:00:35:00', '2016:064:01:20:00'),
                                    ('2021:059:17:20:00', '2021:059:17:40:00'),
                                    ('2021:104:11:14:00', '2021:104:11:25:00')])
    for msid in msids:
        years = CxoTime(dat[msid].times).decimalyear
        yrs_bin, vals_bin = binned_median(years, dat[msid].vals)
        setattr(dat[msid], 'vals_smooth', np.interp(x=years, xp=yrs_bin, fp=vals_bin))

    manvrs = events.manvrs.table
    manvrs.add_index('obsid')

    result = Table()
    result['obsid'] = np.unique(obsids)
    for msid in msids:
        result[msid] = np.nan
        result[f'{msid}_val'] = np.nan
        result[f'{msid}_smooth'] = np.nan

    for row in result:
        try:
            manvr = manvrs.loc[row["obsid"]]
        except KeyError:
            logging.info(f'No manvr for {row["obsid"]}')
            continue
        if isinstance(manvr, Table):
            manvr = manvr[0]
        tstart = CxoTime(manvr['manvr_stop']).secs
        tstop = CxoTime(manvr['next_manvr_start']).secs
        for msid in msids:
            i0, i1 = np.searchsorted(dat[msid].times, (tstart, tstop))
            # perhaps we should check time earlier
            if i0 == i1:
                continue
            row[f'{msid}'] = np.mean(dat[msid].vals[i0:i1] - dat[msid].vals_smooth[i0:i1])
            row[f'{msid}_val'] = np.mean(dat[msid].vals[i0:i1])
            row[f'{msid}_smooth'] = np.mean(dat[msid].vals_smooth[i0:i1])
    result.add_index('obsid')
    return result
